/*
Downloads a csv of the list of First and Last names of profiles on screen
Later use =concat formula to prepare the email address format
*/
function extractAndDownloadCSV() {

  const nameElements = document.querySelectorAll('.lt-line-clamp--single-line');

  const rows = [];

  nameElements.forEach(el => {

    let fullName = el.innerText.trim();

    // Remove certifications after comma
    fullName = fullName.split(",")[0];

    const parts = fullName.split(/\s+/);

    const firstName = parts[0] || "";
    const lastName = parts.slice(1).join(" ") || "";

    rows.push(`${firstName},${lastName}`);

  });

  const csvContent = "First Name,Last Name\n" + rows.join("\n");

  // Create downloadable file
  const blob = new Blob([csvContent], { type: "text/csv" });
  const url = URL.createObjectURL(blob);

  const a = document.createElement("a");
  a.href = url;
  a.download = "linkedin_names.csv";
  document.body.appendChild(a);
  a.click();

  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  console.log("CSV downloaded with", rows.length, "names");

}

extractAndDownloadCSV();
