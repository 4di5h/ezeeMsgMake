function extractAndDownloadCSV() {
  // Use XPath to get text nodes from name elements
  const xpathResult = document.evaluate(
    '//*[contains(@class, "lt-line-clamp--single-line") and contains(@class, "t-black")]/text()',
    document,
    null,
    XPathResult.ORDERED_NODE_SNAPSHOT_TYPE,
    null
  );

  const rows = [];
  const seen = new Set();

  for (let i = 0; i < xpathResult.snapshotLength; i++) {
    let fullName = xpathResult.snapshotItem(i).textContent.trim();
    if (!fullName || fullName.length < 2) continue;

    // Remove certifications/suffixes after comma
    fullName = fullName.split(",")[0].trim();

    // Deduplicate
    if (seen.has(fullName)) return;
    seen.add(fullName);

    const parts = fullName.split(/\s+/);
    const firstName = parts[0] || "";
    const lastName = parts.slice(1).join(" ") || "";

    rows.push(`"${firstName}","${lastName}"`);
  }

  if (rows.length === 0) {
    console.warn("No names found.");
    return;
  }

  const csvContent = "First Name,Last Name\n" + rows.join("\n");
  const blob = new Blob([csvContent], { type: "text/csv" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "linkedin_names.csv";
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  console.log(`CSV downloaded with ${rows.length} names`);
}
extractAndDownloadCSV();
