function extractAndDownloadCSV() {
  const xpathResult = document.evaluate(
    '/html/body/div[6]/div[3]/div/div[2]/div/div[2]/main/div[2]/div/div/div[2]/div/div[1]/ul/li/div/section/div/div/div[2]/div[1]/a/div',
    document,
    null,
    XPathResult.ORDERED_NODE_SNAPSHOT_TYPE,
    null
  );

  console.log("Nodes found:", xpathResult.snapshotLength);

  const rows = [];
  const seen = new Set();

  for (let i = 0; i < xpathResult.snapshotLength; i++) {
    let fullName = xpathResult.snapshotItem(i).innerText.trim();
    if (!fullName || fullName.length < 2) continue;

    fullName = fullName.split(",")[0].trim();
    if (seen.has(fullName)) continue;
    seen.add(fullName);

    const parts = fullName.split(/\s+/);
    const firstName = parts[0] || "";
    const lastName = parts.slice(1).join(" ") || "";
    rows.push(`"${firstName}","${lastName}"`);
  }

  if (rows.length === 0) {
    console.warn("No names found. Check if the XPath structure changed.");
    return;
  }

  const csvContent = "First Name,Last Name\n" + rows.join("\n");
  const blob = new Blob([csvContent], { type: "text/csv" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "linkedin_names.csv";
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  console.log(`CSV downloaded with ${rows.length} names`);
}
extractAndDownloadCSV();
