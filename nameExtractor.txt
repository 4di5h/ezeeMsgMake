function extractAndDownloadCSV() {
  const elements = document.querySelectorAll('div[class*="lt-line-clamp--single-line"][class*="t-black"]');

  const rows = [];
  const seen = new Set();

  elements.forEach(el => {
    let fullName = el.innerText.trim();
    if (!fullName || fullName.length < 2) return;

    fullName = fullName.split(",")[0].trim();
    if (seen.has(fullName)) return;
    seen.add(fullName);

    const parts = fullName.split(/\s+/);
    const firstName = parts[0] || "";
    const lastName = parts.slice(1).join(" ") || "";
    rows.push(`"${firstName}","${lastName}"`);
  });

  if (rows.length === 0) {
    console.warn("No names found.");
    return;
  }

  const csvContent = "First Name,Last Name\n" + rows.join("\n");
  const blob = new Blob([csvContent], { type: "text/csv" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "linkedin_names.csv";
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  console.log(`CSV downloaded with ${rows.length} names`);
}
extractAndDownloadCSV();
